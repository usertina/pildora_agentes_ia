class AgentMemory {
    constructor() {
        this.conversationHistory = [];
        this.userPreferences = {};
        this.longTermMemory = [];
    }
    
    // Guardar mensaje en memoria
    remember(message, role = 'user') {
        this.conversationHistory.push({
            role: role,
            content: message,
            timestamp: new Date().toISOString()
        });
        
        // Mantener solo los Ãºltimos 10 mensajes
        if (this.conversationHistory.length > 10) {
            this.conversationHistory.shift();
        }
    }
    
    // Obtener contexto para el agente
    getContext() {
        return this.conversationHistory.map(msg => 
            `${msg.role}: ${msg.content}`
        ).join('\n');
    }
    
    // Guardar preferencias del usuario
    savePreference(key, value) {
        this.userPreferences[key] = value;
    }
    
    // Obtener preferencias
    getPreference(key) {
        return this.userPreferences[key] || null;
    }
    
    // Buscar en memoria a largo plazo
    searchMemory(query) {
        return this.longTermMemory.filter(item => 
            item.content.toLowerCase().includes(query.toLowerCase())
        );
    }
}
