class EmailAgent:
    def __init__(self):
        self.memory = []  # Memoria del agente
        self.tools = {
            'analyze': self.analyze_email,
            'notify': self.send_notification,
            'archive': self.archive_email
        }
    
    def think(self, email):
        """Paso 1: Pensar/Analizar"""
        print(f"üß† Analizando email de: {email.sender}")
        
        # Analiza el contenido
        is_important = self.analyze_email(email)
        
        return is_important
    
    def decide(self, analysis_result):
        """Paso 2: Decidir"""
        if analysis_result['priority'] == 'high':
            return 'notify'
        elif analysis_result['priority'] == 'medium':
            return 'save_for_review'
        else:
            return 'archive'
    
    def act(self, decision, email):
        """Paso 3: Actuar"""
        print(f"‚öôÔ∏è Decisi√≥n: {decision}")
        
        action = self.tools[decision]
        action(email)
    
    def run(self, email):
        """Ejecutar el ciclo completo"""
        analysis = self.think(email)
        decision = self.decide(analysis)
        self.act(decision, email)
        
        print("‚úÖ Tarea completada")

class AgentMemory {
    constructor() {
        this.conversationHistory = [];
        this.userPreferences = {};
        this.longTermMemory = [];
    }
    
    // Guardar mensaje en memoria
    remember(message, role = 'user') {
        this.conversationHistory.push({
            role: role,
            content: message,
            timestamp: new Date().toISOString()
        });
        
        // Mantener solo los √∫ltimos 10 mensajes
        if (this.conversationHistory.length > 10) {
            this.conversationHistory.shift();
        }
    }
    
    // Obtener contexto para el agente
    getContext() {
        return this.conversationHistory.map(msg => 
            `${msg.role}: ${msg.content}`
        ).join('\n');
    }
    
    // Guardar preferencias del usuario
    savePreference(key, value) {
        this.userPreferences[key] = value;
    }
    
    // Obtener preferencias
    getPreference(key) {
        return this.userPreferences[key] || null;
    }
    
    // Buscar en memoria a largo plazo
    searchMemory(query) {
        return this.longTermMemory.filter(item => 
            item.content.toLowerCase().includes(query.toLowerCase())
        );
    }
}
